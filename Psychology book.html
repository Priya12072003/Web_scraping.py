import requests
from bs4 import BeautifulSoup
import pandas as pd
import time

# Headers to mimic a real browser visit
headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko)"
                  " Chrome/90.0.4430.93 Safari/537.36"
}

base_url = "https://www.amazon.in/s?k=psychological+books&crid=1MHOKCXNUMYS&sprefix=books+ps%2Caps%2C393&ref=nb_sb_ss_ts-doa-p_1_8"

# Store results
product_data = []

# Set how many pages to scrape
for page in range(1, 50):  # Scraping first 5 pages
    url = base_url.format(page)
    print(f"Scraping page {page}: {url}")
    
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.content, "html.parser")
    
    results = soup.find_all("div", {"data-component-type": "s-search-result"})
    
    for item in results:
        name = item.h2.text.strip() if item.h2 else None

        try:
            price = item.find("span", class_="a-price-whole").text.replace(",", "")
        except:
            price = None
        
        try:
            rating = item.find("span", class_="a-icon-alt").text
        except:
            rating = None
        
        try:
            reviews = item.find("span", {"class": "a-size-base s-underline-text"}).text  
        except:
            reviews = None

        # Amazon doesn't always show a separate "offer" price - using regular price
        offer = price

        product_data.append({
            "Product Name": name,
            "Price (INR)": price,
            "Rating": rating,
            "Reviews": reviews,
            "Offer": offer
        })
    
    time.sleep(2)  # Respectful crawling

# Convert to DataFrame
df = pd.DataFrame(product_data)

# Export to CSV
df.to_csv("psychology_book.csv5", index=False, encoding="utf-8")
print("Data scraping complete. File saved as 'psychology_book5.csv'")

import pandas as pd
import numpy as np
from scipy.stats import chi2_contingency


df = pd.read_csv("updated_psychology_book.csv")

 #Creating rating category - 'High' if >4.0 else 'Low'
df['Rating Category'] = np.where(df['Rating'] > 4.0, 'High', 'Low')

#Creating offer status - 'Has Offer' if not 0, else 'No Offer'
df['Offer Status'] = np.where(df['Offer'].astype(str) != '0', 'Has Offer', 'No Offer')


contingency_table = pd.crosstab(df['Rating Category'], df['Offer Status'])


chi2_stat, p_val, dof, expected = chi2_contingency(contingency_table)


print("Contingency Table:\n", contingency_table)
print(f"\nChi-square Statistic: {chi2_stat}")
print(f"P-value: {p_val}")

if p_val < 0.05:
    print("\n There is a significant association between rating and offer.")
else:
    print("\n No significant association between rating and offer.")

import pandas as pd
import numpy as np
from scipy.stats import f_oneway, pearsonr


df = pd.read_csv("updated_psychology_book.csv")


df['Price (INR)'] = pd.to_numeric(df['Price (INR)'], errors='coerce')
df = df.dropna(subset=['Price (INR)', 'Rating'])


df['Rating Category'] = np.where(df['Rating'] > 4.0, 'High', 'Low')


high_group = df[df['Rating Category'] == 'High']['Price (INR)']
low_group = df[df['Rating Category'] == 'Low']['Price (INR)']

anova_stat, anova_p = f_oneway(high_group, low_group)

print(" ANOVA Test:")
print(f"F-statistic = {anova_stat}")
print(f"P-value = {anova_p}")
if anova_p < 0.05:
    print(" Significant difference in prices between groups.\n")
else:
    print(" No significant difference in prices between groups.\n")


correlation_stat, correlation_p = pearsonr(df['Price (INR)'], df['Rating'])

print(" Correlation Test:")
print(f"Correlation Coefficient = {correlation_stat}")
print(f"P-value = {correlation_p}")
if correlation_p < 0.05:
    print(" Significant correlation between price and rating.")
else:
    print(" No significant correlation between price and rating.")
